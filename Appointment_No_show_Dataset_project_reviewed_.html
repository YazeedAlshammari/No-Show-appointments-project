# Calculate the time gap between scheduling and appointment dates in days
# 'AppointmentDay' and 'ScheduledDay' are in string format
df['AppointmentDay'] = pd.to_datetime(df['AppointmentDay'])
df['ScheduledDay'] = pd.to_datetime(df['ScheduledDay'])

# Calculate the time gap between scheduling and appointment dates in days
df['TimeGap'] = (df['AppointmentDay'] - df['ScheduledDay']).dt.days

# Create a scatter plot
plt.figure(figsize=(10, 6))
plt.scatter(df['TimeGap'], np.where(df['No-show'] == 'No', 1, 0), alpha=0.5, marker='o')

# Customize the plot
plt.title('Effect of Time Gap on Attendance')
plt.xlabel('Time Gap (Days)')
plt.ylabel('Attendance (1: Show, 0: No-show)')

# Display the plot
plt.show()

# Calculate the no-show rate for each group
no_show_sms_received = df[df['SMS_received'] == 1]['No-show'].value_counts(normalize=True) * 100
no_show_no_sms_received = df[df['SMS_received'] == 0]['No-show'].value_counts(normalize=True) * 100

# Create a bar plot
fig, ax = plt.subplots(figsize=(8, 6))
bar_width = 0.35
index = np.arange(len(no_show_sms_received))

bar1 = ax.bar(index, no_show_sms_received, bar_width, label='SMS Received')
bar2 = ax.bar(index + bar_width, no_show_no_sms_received, bar_width, label='No SMS Received')

# Customize the plot
ax.set_xlabel('Appointment Attendance')
ax.set_ylabel('Percentage')
ax.set_title('Effect of SMS Reminders on No-show Rate')
ax.set_xticks(index + bar_width / 2)
ax.set_xticklabels(no_show_sms_received.index)
ax.legend()

# Display the plot
plt.show()
